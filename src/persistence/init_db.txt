from sqlalchemy import create_engine
from src.persistence.models import Base

# URL directa (sin pasar por configparser que es el que falla con el Unicode)
DATABASE_URL = "postgresql+psycopg2://admin:password@localhost:5432/eka_db"

def create_tables():
    engine = create_engine(DATABASE_URL)
    Base.metadata.create_all(engine)
    print("Tablas creadas exitosamente usando SQLAlchemy nativo.")

if __name__ == "__main__":
    create_tables()
